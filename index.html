Code à Copier

Voici le script que vous pouvez copier et utiliser :



const links = Array.from(document.querySelectorAll('a')).map(a => a.href);
const htmlContent = '<!DOCTYPE html><html lang="fr"><head><meta charset="UTF-8"><meta name="viewport" content="width=device-width, initial-scale=1.0"><title>Liens</title></head><body><h1>' + window.location.hostname + '</h1>' + links.map(link => '<p><a href="' + link + '">' + link + '</a></p>').join('') + '</body></html>';
const blob = new Blob([htmlContent], { type: 'text/html' });
const a = document.createElement('a');
a.href = URL.createObjectURL(blob);
a.download = 'liens.html';
a.click();
URL.revokeObjectURL(a.href);
